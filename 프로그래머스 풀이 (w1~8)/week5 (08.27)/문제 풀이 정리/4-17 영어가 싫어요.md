function solution(numbers) {
    return numbers.match(/(zero|one|two|three|four|five|six|seven|eight|nine)/g)
        .reduce((acc, v) =>{
            switch(v){
                case 'zero':
                    return acc*10 + 0;
                case 'one':
                    return acc*10 + 1;
                case 'two':
                    return acc*10 + 2;
                case 'three':
                    return acc*10 + 3;
                case 'four':
                    return acc*10 + 4;
                case 'five':
                    return acc*10 + 5;
                case 'six':
                    return acc*10 + 6;
                case 'seven':
                    return acc*10 + 7;
                case 'eight':
                    return acc*10 + 8;
                case 'nine':
                    return acc*10 + 9;
            }        
        }, 0);
}

```js
function solution(numbers) {
    const obj = {
        zero: 0, one: 1, two: 2, three: 3, four: 4,
        five: 5, six: 6, seven: 7, eight: 8, nine: 9
    };

    const num = numbers.replace(/zero|one|two|three|four|five|six|seven|eight|nine/g, (v) => {
        return obj[v];
    });

    return Number(num);
}
```

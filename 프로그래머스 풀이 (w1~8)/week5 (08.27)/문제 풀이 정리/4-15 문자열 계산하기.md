```js
function solution(my_string) {
    let method = '+';
    return my_string.split(' ').reduce((acc, v)=>{
        if(Number.isNaN(parseInt(v))){
            method=v;
            return acc;
        }
        else{
            if(method==='+')return acc + parseInt(v);
            if(method==='-')return acc - parseInt(v);
        }
    }, 0);
}
```
